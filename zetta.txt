package com.example.practical1
import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent
import android.content.IntentFilter
import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.widget.Toast
class MainActivity : AppCompatActivity() {
lateinit var receiver: BroadcastReceiver
override fun onCreate(savedInstanceState: Bundle?) {
super.onCreate(savedInstanceState)
setContentView(R.layout.activity_main)
//To launch BroadcastReceiver(),use IntentFilter()
val filter=IntentFilter()
//Adding Actions - POWER_CONNECTED and
POWER_DISCONNECTED
filter.addAction(Intent.ACTION_POWER_CONNECTED)
filter.addAction(Intent.ACTION_POWER_DISCONNECTED)
//BroadcastReceiver() object - receiver
receiver=object:BroadcastReceiver()
{
override fun onReceive(context: Context?, intent: Intent?) {
//Action received through - Toast
Toast.makeText(context,"Charger Connection
Chnaged",Toast.LENGTH_LONG).show()
}
}
//Invoking registerReceiver()
registerReceiver(receiver,filter)
}
//Invoking unregisterReceiver()
override fun onDestroy() {
unregisterReceiver(receiver)
super.onDestroy()
}
}

package com.example.practical4listview
import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.view.Gravity
import android.widget.ArrayAdapter
import android.widget.ListView
import android.widget.Toast
class MainActivity : AppCompatActivity() {
override fun onCreate(savedInstanceState: Bundle?) {
super.onCreate(savedInstanceState)
setContentView(R.layout.activity_main)
//Variables
val countries= arrayOf<String>("India", "Japan","New
Zealand","Australia")
val listview=findViewById<ListView>(R.id.lstVMyList)
//Creating a ListView adapter for listview variable to bundle data
//simple_list_item_1 (Its a raw layout by ListView) - By default, xml
document which binds data (ie countries) to ListView
listview.adapter=
ArrayAdapter<String>(this,android.R.layout.simple_list_item_1,countries)
//Setting setOnItemClickListener to Toast a message on click of item
listview.setOnItemClickListener { parent, view, position, id ->
val toast= Toast.makeText(this,"You have clicked on - \n" +
countries[position],Toast.LENGTH_LONG)
toast.setGravity(Gravity.CENTER,0,0)
toast.show()
}
}
}

package com.example.practical5

super.onCreate(savedInstanceState)
setContentView(R.layout.activity_main)
}
override fun onCreateOptionsMenu(menu: Menu?): Boolean {
//Binds menu.xml file to menu bar of your app bar
menuInflater.inflate(R.menu.menu,menu)
return super.onCreateOptionsMenu(menu)
}
override fun onOptionsItemSelected(item: MenuItem): Boolean {
//Binds data to menu.xml file
val id = item.itemId
if(id==R.id.search)
{
Toast.makeText(this, "Search", Toast.LENGTH_LONG).show()
}
if(id==R.id.aboutus)
{
Toast.makeText(this, "About Us", Toast.LENGTH_LONG).show()
}
if(id==R.id.contactus)
{
Toast.makeText(this, "Contact Us", Toast.LENGTH_LONG).show()
}
return super.onOptionsItemSelected(item)
}
}

package com.example.practical6
import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.support.design.widget.BottomNavigationView
import android.support.design.widget.NavigationView
import android.support.v4.app.Fragment
import kotlinx.android.synthetic.main.activity_main.*
class MainActivity : AppCompatActivity() {
override fun onCreate(savedInstanceState: Bundle?) {
super.onCreate(savedInstanceState)
setContentView(R.layout.activity_main)
navigation.setOnNavigationItemSelectedListener
(mOnNavigationItemSelectedListener)
}
private val mOnNavigationItemSelectedListener =
BottomNavigationView.OnNavigationItemSelectedListener {
item->
when(item.itemId)
{
R.id.videos->
{
val videosFragment=VideosFragment()
openFragment(videosFragment)
return@OnNavigationItemSelectedListener true
}
R.id.notes->
{
val notesFragment=NotesFragement()
openFragment(notesFragment)
return@OnNavigationItemSelectedListener true
}
R.id.quiz->
{
val quizFragment=QuizFragement()
openFragment(quizFragment)
return@OnNavigationItemSelectedListener true
}
}//when
false
}//mOnNavigationItemSelectedListener
private fun openFragment(fragment: Fragment)
{
val transaction=supportFragmentManager.beginTransaction()
transaction.replace(R.id.container,fragment)
transaction.addToBackStack(null)
transaction.commit()
}//openFragment
}//class MainActivity

}//onCreate
}

package com.example.practical8
import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.content.Context
import android.content.Intent
import android.graphics.Color
import android.os.Build
import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
class MainActivity : AppCompatActivity() {
//Variables
lateinit var notificationManager : NotificationManager
lateinit var notificationChannel: NotificationChannel
lateinit var builder:Notification.Builder
//Declare channel id
val channelId="package com.example.practical8"
val description="My Notification"
override fun onCreate(savedInstanceState: Bundle?) {
super.onCreate(savedInstanceState)
setContentView(R.layout.activity_main)
//Binding Notification to the button
val show=findViewById<Button>(R.id.btnShowNotification)
//Create a NotificationManager and use Service
notificationManager=getSystemService(Context.NOTIFICATION_SERVICE)
as NotificationManager
show.setOnClickListener{
val intent=Intent(applicationContext,MainActivity::class.java)
//Create PendingIntent() - gives tokens to other applications to execute
this program
val
pendingIntent=PendingIntent.getActivity(this,0,intent,PendingIntent.FLAG_UP
DATE_CURRENT)
if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
notificationChannel=
NotificationChannel(channelId,description,NotificationManager.IMPORTANCE
_HIGH)
//Configure the lights of the device
notificationChannel.enableLights(true)
//Configure light color
notificationChannel.lightColor=Color.RED
//Configure vibration mode
notificationChannel.enableVibration(true)
//Create notification
notificationManager.createNotificationChannel(notificationChannel)
//Build notification
builder=Notification.Builder(this, channelId)
//Set the contents of notification
.setContentTitle("Android")
.setContentText("New Message")
.setSmallIcon(R.mipmap.ic_launcher)
.setContentIntent(pendingIntent)
}//if
else
{
builder=Notification.Builder(this)
//Set the contents of notification
.setContentTitle("Android")
.setContentText("New Message")
.setSmallIcon(R.mipmap.ic_launcher)
.setContentIntent(pendingIntent)
}//else
notificationManager.notify(0,builder.build())
}//setOnClickListener
}//onCreate
}
